#购买slot1的商品
uma_buy_effect_slot_1 = {
    potential = {
        always = yes
    }
    allow = {
        custom_tooltip = {
            fail_text = uma_you_have_bought_this_slot
            NOT = { has_country_flag = uma_slot_1_bought }
            check_variable = {
                which = uma_num_of_coin
                value >= uma_price_of_goods_1_multiply  #记住，一定是>=，只用=的话，只会在==的时候判定为真。
            }
        }
    }
    effect = {
        hidden_effect = {
        uma_buy_slot_1_goods_effect = yes #封装效果，买下槽位1的东西
            subtract_variable = {
                which = uma_num_of_coin
                value = uma_price_of_goods_1_multiply #不要直接引用变量（energy = 变量），而是要用base=1，乘以这个变量
            }
            set_country_flag = uma_slot_1_bought
        }
    }
}

    potential = {
        always = yes
    }
    allow = {
        custom_tooltip = {
            fail_text = uma_you_have_bought_this_slot
            NOT = { has_country_flag = uma_slot_6_bought }
            check_variable = {
                which = uma_num_of_coin
                value >= uma_price_of_goods_6_multiply  #记住，一定是>=，只用=的话，只会在==的时候判定为真。
            }
        }
    }
    effect = {
        hidden_effect = {
        uma_buy_slot_6_goods_effect = yes #封装效果，买下槽位1的东西
            subtract_variable = {
                which = uma_num_of_coin
                value = uma_price_of_goods_6_multiply #不要直接引用变量（energy = 变量），而是要用base=1，乘以这个变量
            }
            set_country_flag = uma_slot_6_bought
        }
    }
