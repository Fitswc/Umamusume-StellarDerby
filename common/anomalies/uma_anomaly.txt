 uma_bourbon_anomaly = {

    desc = "uma_bourbon_anomaly_desc"
    should_ai_use = no
    picture = GFX_evt_baol_ruins
    level = 7
    null_spawn_chancec = 0
    max_once_global = yes
    spawn_chance = {
        base = 0
        modifier = {
            add = 8
			is_planet_class = pc_tropical

			id = uma_bour_arc.1 #生成时调用

		}
	}

        1 = {
            max_once = yes

            anomaly_event = uma_bour_arc.2
        }
    }

}
