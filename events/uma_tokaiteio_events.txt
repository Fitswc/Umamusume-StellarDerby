country_event = {
	id = uma_tokaiteio_event.210
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		if = {
			limit = {
				check_variable = {
					which = tokaiteio_choice_two
					value >= 3
				}
			}
			country_event = {
				id = uma_tokaiteio_event.300
			}
		}
		if = {
			limit = {
				AND = {
					check_variable = {
						which = tokaiteio_choice_two
						value < 3
					}
					check_variable = {
						which = tokaiteio_choice_two
						value >= 1
					}
				}
			}
			country_event = {
				id = uma_tokaiteio_event.301
			}
		}
		if = {
			limit = {
				check_variable = {
					which = tokaiteio_choice_two
					value = 0
				}
			}
			country_event = {
				id = uma_tokaiteio_event.302
			}
		}
	}

	after = {
		if = {
			limit = {
				check_variable = {
					which = tokaiteio_choice_one
					value >= 4
				}
			}
			set_country_flag = uma_teio_become_governor
		}
		else_if = {
			limit = {
				check_variable = {
					which = tokaiteio_choice_two
					value >= 4
				}
			}
			set_country_flag = uma_teio_become_admiral
		}
		else_if = {
			limit = {
				check_variable = {
					which = tokaiteio_choice_three
					value >= 4
				}
			}
			set_country_flag = uma_teio_become_scientist
		}
		else = {
			random_list = {
				33 = {
					set_country_flag = uma_teio_become_governor
					modifier = {
						factor = 2
						root = {
							check_variable = {
								which = tokaiteio_choice_one
								value > 2
							}
						}
					}
				}
				33 = {
					set_country_flag = uma_teio_become_admiral
					modifier = {
						factor = 2
						root = {
							check_variable = {
								which = tokaiteio_choice_two
								value > 2
							}
						}
					}
				}
				33 = {
					set_country_flag = uma_teio_become_scientist
					modifier = {
						factor = 2
						root = {
							check_variable = {
								which = tokaiteio_choice_three
								value > 2
							}
						}
					}
				}
			}
		}
	}
}
