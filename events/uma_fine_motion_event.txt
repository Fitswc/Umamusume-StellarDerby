#Trigger(On First Contact)
first_contact_event = {
	id = uma_finemotion.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = {
			NOT = {
				has_country_flag = uma_fine_motion_happened
			}
		}
	}


	immediate = {
		from = {
			country_event = {
				id = uma_finemotion.2
				days = 300
			}
			set_country_flag = uma_fine_motion_happened
		}
	}
}
